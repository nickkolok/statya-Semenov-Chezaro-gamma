\section{N+1. Название следующего параграфа}


\begin{RTheorem}
	\label{thm:dotnumber}
	Теоремы нумеруются в виде N.K, где N --- номер параграфа,
	K --- порядковый номер теоремы в нём.
\end{RTheorem}

\begin{RTheorem*}
	Некоторые теоремы не имеют номера.
\end{RTheorem*}

\begin{RTheorem}[об инвариантности]
	\label{invariance}
	Теоремы, не имеющие номера, не влияют на нумерацию прочих теорем.
\end{RTheorem}

\begin{Rtheorem}
	\label{thm:unindented}
	Для длинных списков теорем можно использовать окружения без вертикального отступа.
\end{Rtheorem}

\begin{Rtheorem}
	\label{thm:unindented2}
	Для длинных списков теорем можно использовать окружения без вертикального отступа.
\end{Rtheorem}

\begin{RLemma}
	\label{thm:lemma1}
	Леммы нумеруются отдельно от теорем.
\end{RLemma}

\begin{RLemma}
	На леммы и теоремы можно ссылаться по меткам, как на уравнения.
	Это очень полезно при внесении изменений.
\end{RLemma}

\begin{proof}
	Достаточно посмотреть, например, на теоремы~\ref{thm:dotnumber},
	\ref{invariance} и \ref{thm:unindented2},
	а также лемму~\ref{thm:lemma1}.
\end{proof}
