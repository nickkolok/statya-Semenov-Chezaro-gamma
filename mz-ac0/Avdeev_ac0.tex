\end{keywords}


\footnotetext{Работа выполнена в Воронежском университете при поддержке РНФ, грант 16-11-10125.}

Через $\ell_\infty$ будем обозначать пространство ограниченных последовательностей с обычной нормой
\begin{equation*}
	\|x\| = \max_{k\in\mathbb{N}} |x_k|
	,
\end{equation*}
через $c$ и $c_0$~--- пространства сходящихся и сходящихся к нулю последовательностей соотв.,
через $\mathbb{N}$~--- множество натуральных чисел,
через $T$~--- оператор сдвига влево, т.е.
$T(x_1,x_2,...) = (x_2,x_3,...)$.


Изучая множество банаховых пределов, Лоренц \cite{Lorentz}
ввёл пространство почти сходящихся последовательностей $ac$.
Последовательность $x\in\ell_\infty$ называется почти сходящейся,
если существует такое число $t$, что
\begin{equation}
\label{eq:intro_ac}
	\lim_{n\to\infty} \frac{1}{n} \sum_{k=m+1}^{m+n} x_k = t
\end{equation}
равномерно по $m$.
Если $t=0$, то говорят, что последовательность
принадлежит пространству $ac_0$ почти сходящихся к нулю последовательностей.
Легко видеть, что $x\in ac_0$ тогда и только тогда, когда
\begin{equation}\label{crit_pos_ac0}
	\forall(B  \in\mathbb{N})
	\exists(n_0\in\mathbb{N})
	\exists(m_0\in\mathbb{N})
	\forall(n\geq n_0)
	\forall(m\geq m_0)
	\\
	\left[
		\left|
		\frac{1}{n}
		\sum_{k=m+1}^{m+n} x_k
		\right|
		<
		\frac{1}{B}
	\right]
	.
\end{equation}
Удобство соотношения \eqref{crit_pos_ac0} состоит в
выборе $m_0$ в зависимости от $B$.
В \cite[теорема 7 и следствие 12]{Alekhno_ac0} доказано, что пространство $ac_0$ замкнуто, недополняемо в $\ell_\infty$ и несепарабельно.
В силу соотношения $ac = ac_0 \oplus \mathbb{R}^1$ этими же тремя свойствами обладает $ac$.

В первой части настоящей статьи приводится критерий почти сходимости неотрицательной последовательности к нулю,
основанный на идеях \cite[\S 5]{SemSukUsa14}.
Во второй части работы рассматривается на $\ell_\infty$ функция
\begin{equation*}
	\alpha(x) = \limsup_{i\to\infty} \max_{i \leq j \leq 2i} |x_i-x_j|
\end{equation*}
и изучается её связь с расстоянием до пространства $c$ для почти сходящихся последовательностей,
а также приводится критерий сходимости почти сходящейся последовательности, основанный на этой функции.



\begin{lemma}
Пусть $n_i$~--- строго возрастающая последовательность чисел из $\mathbb{N}$,
\begin{equation}
	x_k = \left\{\begin{array}{ll}
		1, & \mbox{~если~} k = n_i
		\\
		0  & \mbox{~иначе.~}
	\end{array}\right.
\end{equation}
Для того, чтобы $x\in ac_0$,
необходимо и достаточно, чтобы
\begin{equation}\label{lim_M(j)/j}
	\lim_{j \to \infty} \frac{M(j)}{j} = \infty
	,
\end{equation}
где
\begin{equation*}
	%\quad\mbox{~где~\quad}
	M(j) = \liminf_{i\to\infty} (n_{i+j} - n_i)
	.
\end{equation*}
\end{lemma}

\paragraph{Необходимость}
Понятно, что $M(j)\in\mathbb{N}$
и для любого $j$ существует лишь конечное количество отрезков из $M(j)+1$ элементов,
содержащих более $j+1$ единиц.
Через $E_j$ будем обозначать конец последнего отрезка из $M(j)+1$ элементов,
содержащего более $j+1$ единиц.
Кроме того, существует бесконечное количество отрезков из $M(j)+1$ элементов,
содержащих ровно $j+1$ единиц.
Рассмотрев каждый отрезок без последнего элемента, получим бесконечное количество отрезков из $M(j)$ элементов,
содержащих ровно $j$ единиц.

%\begin{proof}
Предположим противное, т.е.
\begin{equation}
	\liminf_{j \to \infty} \frac{M(j)}{j} = q < \infty
	.
\end{equation}
Очевидно, $q>0$.
Найдётся счётное множество
$J\subset\mathbb{N}$ такое, что
$
	\forall(j\in J)\left[q \leq {M(j)}/{j} \leq Q \right]\!\!,
$
где $Q$~--- наименьшее натуральное число, превосходящее $q$.
Для любого $j\in J$ существует бесконечно много отрезков из $jQ$ элементов,
в каждом из которых содержится не менее $j$ единиц.
%
Т.к. $x\in ac_0$, то
\begin{equation}\label{Lorencz_ac0_epsilon}
	\forall(\varepsilon>0)
	\exists(n_0\in\mathbb{N})
	\forall(n \geq n_0)
	\forall(m\in\mathbb{N})
	\left[
		\frac{1}{n} \sum_{k=m+1}^{m+n}x_k < \varepsilon
	\right]
	.
\end{equation}

Положим $\varepsilon = 1/(Q+1)$ и отыщем $n_0$, удовлетворяющее \eqref{Lorencz_ac0_epsilon}.
Рассмотрим $n\in J$, $n\geq n_0$.
Такое $n$ всегда найдётся, т.к. $J$ счётно и $J\subset\mathbb{N}$.

начинался с $(m+1)$-го элемента.
Тогда
\begin{equation}
	\frac{1}{nQ}\sum_{k=m+1}^{m+nQ}x_k
	\geq
	\frac{1}{nQ} \cdot n
	=
	\frac{1}{Q}
	>
	\frac{1}{Q+1}
	=
	\varepsilon,
\end{equation}
что противоречит \eqref{Lorencz_ac0_epsilon}.
Полученное противоречие  доказывает необходимость.

\paragraph{Достаточность}

Из равенства \eqref{lim_M(j)/j} вытекает, что
\begin{equation}\label{lim_M(j)/j_infty_def}
	\forall(C  \in\mathbb{N})
	\exists(j_0\in\mathbb{N})
	\forall(j \geq j_0)
	\left[
		\frac{M(j)+1}{j+1}>C
	\right]
	.
\end{equation}
Покажем, что выполнено
\eqref{crit_pos_ac0}.
Действительно, зафиксируем $B$.
Используя \eqref{lim_M(j)/j_infty_def} и приняв $C=2B$,
отыщем $j_0$ такое, что для любого $j \geq j_0$ выполнено
$M(j)+1>2B(j+1)$.
Положим $n_0 = 2B(j_0+1)$
и
$$
	m_0 = 2+\max_{1\leq j \leq j_0+1} E_j
	.
$$

Тогда для любых $m\geq m_0$ и $n\geq n_0$ имеем
\begin{multline*}
	\frac{1}{n} \sum_{k=m+1}^{m+n} x_k
	<
	\frac{1}{n} \left( \frac{n}{M(j_0)+1} + 1 \right) (j_0 + 1)
	=
	\frac{j_0+1}{M(j_0)+1} + \frac{j_0+1}{n}
	<%\leq
	\frac{1}{2B} + \frac{j_0+1}{n_0}
	=
	%\frac{1}{2B} + \frac{1}{2B}
	%=
	\frac{1}{B}
	,
\end{multline*}
т.е. условие критерия \eqref{crit_pos_ac0} выполнено
и $x\in ac_0$,
что и требовалось доказать.
Лемма доказана.
%\end{proof}

Определим нелинейный оператор $\lambda$--срезки $A_\lambda$
для $x = (x_1, x_2, ...)\in\ell_\infty$ как
\begin{equation}
	(A_\lambda x)_k = \begin{cases}
		1, & \mbox{~если~} x_k \geq \lambda
		\\
		0  & \mbox{~иначе.~}
	\end{cases}
\end{equation}

\begin{theorem}
\label{thm:lambda_prelim}
Пусть $x\in\ell_\infty$, $x\geq 0$.
Тогда
$
	x\in ac_0
$
если и только если
для любого $\lambda > 0$
выполнено
$
	A_\lambda x \in ac_0
$
.
\end{theorem}

\paragraph{Необходимость}
Пусть $x\in ac_0$.
Зафиксируем $\lambda > 0$.
Пусть $y=A_\lambda x$, тогда
\begin{equation}
	(\lambda y)_k = \begin{cases}
		\lambda, & \mbox{~если~} x_k \geq \lambda
		\\
		0  & \mbox{~иначе~}
	\end{cases}
\end{equation}
Таким образом, $0 \leq \lambda y \leq x$.
Следовательно, если $x \in ac_0$,
то $\lambda y \in ac_0$ и $y \in ac_0$.%,
%что и требовалось доказать.

\paragraph{Достаточность}
Случай $x=0$ тривиален.
Пусть $x\neq 0$.
Т.к.
$
	x\in ac_0
$
тогда и только тогда, когда
$
	{x}/{\|x\|}\in ac_0
$,
то будем полагать $\|x\|\leq 1$.

%Более того,
%$
%	A_\lambda x \in ac_0
%$
%если и только если
%$% \Leftrightarrow
%	(1-\lambda)A_\lambda x \in ac_0
%	.
%$

Предположим противное, т.е. $x\notin ac_0$,
но
\begin{equation}
	\label{A_lambda_x_in_ac0}
	\forall(\lambda>0)[A_\lambda x \in ac_0]
	.
\end{equation}
Тогда
\begin{equation}\label{ac0_lambda_Lorencz_neg}
	\exists(\varepsilon_0 > 0)
	\forall(n_0 \in \mathbb{N})
	\exists(n > n_0)
	\exists(m \in \mathbb{N})
	\left[
		\frac{1}{n}\sum_{k=m+1}^{m+n} x_k \geq \varepsilon_0
	\right]
	.
\end{equation}
Найдём $\varepsilon_0$, удовлетворяющее \eqref{ac0_lambda_Lorencz_neg}.
Положим
$
	\varepsilon = \min\left\{ {\varepsilon_0}/{2}, {1}/{2} \right\}
	.
$
Легко видеть, что
\begin{equation}\label{ac0_lambda_Lorencz_neg_epsilon}
	\forall(n_0 \in \mathbb{N})
	\exists(n > n_0)
	\exists(m \in \mathbb{N})
	\left[
		\frac{1}{n}\sum_{k=m+1}^{m+n} x_k > \varepsilon
	\right]
	.
\end{equation}
В силу предположения~\eqref{A_lambda_x_in_ac0} выполнено
$A_{\varepsilon/2} x \in ac_0$.
Т.к. $\varepsilon < 1$, то $1-\varepsilon/2 > 0$.
Положим $y=(1-\varepsilon/2)A_{\varepsilon/2}x$.
Тогда $y\in ac_0$ и $y \geq 0$.
В силу \eqref{eq:intro_ac}
\begin{equation}\label{ac0_lambda_Lorencz}
	\forall(\varepsilon_1 > 0)
	\exists(n_1 \in \mathbb{N})
	\forall(n_2 > n_1)
	\forall(m_2 \in \mathbb{N})
	\left[
		\frac{1}{n}\sum_{k=m+1}^{m+n} y_k < \varepsilon_1
	\right]
	.
\end{equation}

Возьмём в \eqref{ac0_lambda_Lorencz} $\varepsilon_1 = \varepsilon/2$
и отыщем $n_1$.
Примем в \eqref{ac0_lambda_Lorencz_neg_epsilon}
$n_0 = n_1$ и отыщем $n$ и $m$.
Положим в \eqref{ac0_lambda_Lorencz} $n_2 = n$, $m_2 = m$.
Тогда по \eqref{ac0_lambda_Lorencz_neg_epsilon} и \eqref{ac0_lambda_Lorencz}
соответственно
\begin{equation}
	\frac{1}{n}\sum_{k=m+1}^{m+n} x_k > \varepsilon
	\qquad
	\mbox{и}
	\qquad
	\frac{1}{n}\sum_{k=m+1}^{m+n} y_k < \frac{\varepsilon}{2}
	.
\end{equation}
Вычитая, получим
\begin{equation}
	\frac{1}{n}\sum_{k=m+1}^{m+n} (x_k - y_k) > \frac{\varepsilon}{2}
	.
\end{equation}
Если среднее арифметическое чисел вида $x_k - y_k$ больше $\varepsilon/2$,
то существует хотя бы один индекс $k$ такой, что $x_k - y_k > \varepsilon/2$.
%
Предположим, что $k$ таково, что $x_k < \varepsilon/2$.
Тогда $y_k = 0$ и $x_k - y_k < \varepsilon/2$.
Значит, предположение неверно и $x_k \geq \varepsilon/2$.
Тогда $y_k = 1-\varepsilon/2$ и с учётом $\|x\|\leq 1$ имеем
$
	x_k - y_k \leq 1- y_k = 1 - (1-\varepsilon/2) = \varepsilon/2
	.
$
Следовательно, требуемого индекса $k$ не существует,
и \eqref{ac0_lambda_Lorencz_neg} не выполнено.
Полученное противоречие завершает доказательство теоремы~\ref{thm:lambda_prelim}.
%\endproof

Сформулируем теперь этот критерий в терминах функций $M_{\lambda}(j)$.

\begin{theorem}
Пусть $x\in\ell_\infty$, $x \geq 0$,
\begin{equation*}
	0<\lambda < \limsup_{k\to\infty} x_k
	.
\end{equation*}
Пусть $\{k: x_k \geq \lambda \} = \{n(\lambda,1),n(\lambda,2),...\}$.
Обозначим
\begin{equation}
	M_{\lambda}(j) = \liminf_{i\to\infty} n(\lambda,i+j) - n(\lambda,i)
	.
\end{equation}
Для того, чтобы $x\in ac_0$, необходимо и достаточно, чтобы
для любого $\lambda>0$
\begin{equation}
	\lim_{j \to \infty} \frac{M_{\lambda}(j)}{j} = \infty
	.
\end{equation}
\end{theorem}


\begin{remark}
Требование произвольности $\lambda$ существенно.
Например, для $x_k = 1$ и $\lambda = 2$ имеем $A_\lambda x = 0 \in ac_0$,
но $x\notin ac_0$.
\end{remark}

Пусть $\rho(x,c)$ и $\rho(x,c_0)$~--- расстояния от $x$ до пространств $c$
и $c_0$ соотв.
Легко видеть, что функция $\alpha(x)$ удовлетворяет условию Липшица с константой 2
и для $x\in c$ выполнено
$\alpha(x)=0$.

\begin{lemma}
\label{thm:alpha_x_leq_2_rho_x_c}
	Для любого $x\in\ell_\infty$
	выполнено неравенство
	$
		\alpha(x) \leq 2\rho(x, c)
	$.
\end{lemma}
\begin{proof}
	Пусть $\beta = \rho(x,c)$,
	т.е.
	\begin{equation*}
		\beta = \inf_{y\in c}\|x-y\|
		.
	\end{equation*}
	Тогда
	$
		\forall(\varepsilon > 0)
		\exists(y_\varepsilon \in c)
		\bigl[\|x-y_\varepsilon\| < \beta + \varepsilon\bigr]
		.
	$
	Поскольку $y_\varepsilon \in c$, то $\alpha(y_\varepsilon)=0$.
	Значит,
	\begin{equation*}
		\alpha(x) = |\alpha(x)-\alpha(y_\varepsilon)| \leq 2 \|x-y_\varepsilon\| < 2\beta+2\varepsilon = 2\rho(x,c)+2\varepsilon
		.
	\end{equation*}
	Итак,
	$
		\forall(\varepsilon > 0)[\alpha(x) \leq 2 \rho(x,c)+2\varepsilon]
		,
	$
	откуда немедленно получаем, что $\alpha(x)\leq 2\rho(x,c)$.
	Лемма доказана.
\end{proof}


Оценка леммы~\ref{thm:alpha_x_leq_2_rho_x_c} точна.
Как выясняется, верна и противоположная оценка.

\begin{lemma}
\label{thm:rho_x_c_leq_alpha_t_s_x}
	Для любого $x\in ac$ и для любого натурального $s$ верно
	$
		\rho(x,c)\leq \alpha(T^s x)
	$,
%	где $T$~--- оператор сдвига влево, т.е.
%	$T(x_1,x_2,...) = (x_2,x_3,...)$.
\end{lemma}
\begin{proof}
Зафиксируем $s$.
Пусть $x\in ac$, $\alpha(T^s x)=\varepsilon$.
Т.к. $x\in ac$, то в силу \eqref{eq:intro_ac} существует такое число $t$,
что
\begin{equation}
	\forall(p\in\mathbb{N})
	\exists(n_p \in\mathbb{N})\forall(n>n_p)\forall(m\in\mathbb{N})
	\left[
		\left|
			\frac{1}{n}\sum_{k=m+1}^{m+n} x_k
			-t
		\right|
		<\frac{\varepsilon}{p}
	\right]
	.
\end{equation}

Так как
\begin{equation}
	\label{eq:alpha_Ts_x}
	\alpha(T^s x) = \limsup_{k\to\infty} \max_{k<j\leq 2k-s} |x_k-x_j| = \varepsilon
	,
\end{equation}
то
\begin{equation}
	\forall(p\in\mathbb{N})
	\exists(k_p \in\mathbb{N})\forall(k>k_p)
	\forall(j: k< j \leq 2k-s)
	\left[
		|x_k - x_j|<\varepsilon + \frac{\varepsilon}{p}
	\right]
	.
\end{equation}
Положив $q_p = \max\{n_p, k_p\}$, получим, что для любого $q>q_p$
среднее арифметическое чисел $x_q$, $x_{q+1}$, ... , $x_{2q-s}$ отличается от $t$
не более, чем на $\varepsilon/p$, причём разница любых двух из этих чисел меньше $\varepsilon + \varepsilon/p$.
Следовательно, любое из этих чисел отличается от $t$ менее, чем на $\varepsilon + 2\varepsilon/p$.
В частности,
$
	|x_q - t| < \varepsilon + {2\varepsilon}/{p}
$.
Таким образом,
\begin{equation}
	\forall(p\in\mathbb{N})
	\exists(q_p \in\mathbb{N})
	\forall(q>q_p)
	\left[
		|x_q - t| < \varepsilon + \frac{2\varepsilon}{p}
	\right]
	,
\end{equation}
откуда немедленно следует, что $\rho(x,c) \leq \varepsilon$.
Лемма доказана.
\end{proof}


В силу~\eqref{eq:alpha_Ts_x} последовательность $\alpha(T^s x)$ монотонно невозрастает по $s$,
оставаясь при этом неотрицательной, следовательно, имеет предел.
Из этого наблюдения, а также лемм \ref{thm:alpha_x_leq_2_rho_x_c} и \ref{thm:rho_x_c_leq_alpha_t_s_x}
незамедлительно следует
\begin{theorem}
%\label{thm:rho_x_c_leq_alpha_t_s_x}
	Для любого $x\in ac$
	\begin{equation}
		\frac{1}{2} \alpha(x) \leq \rho(x,c)\leq \lim_{s\to\infty} \alpha(T^s x) \leq \alpha(x)
		.
	\end{equation}
\end{theorem}

\begin{corollary}
	Для любого $x\in ac_0$
	\begin{equation}
		\frac{1}{2} \alpha(x) \leq \rho(x,c_0)\leq \lim_{s\to\infty} \alpha(T^s x) \leq \alpha(x)
		.
	\end{equation}
\end{corollary}
Все оценки точны, т.е. на некоторых элементах достигается равенство.
\begin{corollary}
	Пусть $x\in ac$.
	Тогда $x\in c$ если и только если $\alpha(x) = 0$.
\end{corollary}

\begin{remark}
	Условие $x\in ac$ в лемме \ref{thm:rho_x_c_leq_alpha_t_s_x} существенно.
	Так, в \cite{VZMS} для любого $p\in\mathbb{N}$ строится элемент $x=Sy\notin ac$,
	для которого $\alpha(x) = 1/p$ и $\rho(x,c)=1/2$.
\end{remark}
