Рассмотрим матричную задачу Штурма--Лиувиля
\begin{equation}
\biggl(-ih\frac{d}{dx}\biggr)^2y+A(x)y=Ey,
\qquad x\in\mathbb R^1,
\qquad y(x)\in L_2(\mathbb R^1),
\label{1}
\end{equation}
где $y\in\mathbb C^n$, $h\in[0,1)$ является
малым параметром. Пусть $A(x)=A(x)^\ast$
является $(n\times n)$-матричной функцией из
класса $C^\infty(\mathbb R^1)$.
\begin{condition}
\label{cond1} Пусть $A(x)$ есть матрица
вида
$$
A(x)=\mathbf
U(x)\|\lambda_j(x)\delta_{ij}\| \mathbf
U^\ast(x),
$$
где $\mathbf U(x)$~--
унитарная бесконечно дифференцируемая
матрица и $\lambda_j(x)\in
C^\infty(\mathbb R^1)$~-- вещественные
собственные значения матрицы~$A(x)$.
Обозначим через $D$ множество
точек~$x$, в которых собственные
значения меняют кратность.
\end{condition}

\begin{lemma}
\label{lemm1} $1^\circ$ Пусть $y_j^+$,
$j=1,2$, будут решениями
системы~\eqref{1}, имеющими на левом
интервале $(x_0-\delta<x<x_0)$
асимптотики со знаком ``$+$'',
амплитудой $\rho_{j}^+$ и
начальными данными
$\rho_{j}^+(0)$. Тогда
существует унитарная $(2\times
2)$-матрица~$u_{}$ такая, что на
правом интервале $(x_0<x<x_0+\delta)$
решения~$y_j^+$, $j=1,2$, имеют
асимптотики
\begin{equation}
y_j^+=\sum_{k=1}^2
\exp\biggl\{\frac{i}{h}S_k(x)\biggr\}
\biggl[\frac{e_k(x)\rho_{kj}^+(x)}{\sqrt{p_k(x)}}
+h\psi_{kj}^+(x)+O(h^2)\biggr].
\label{8}
\end{equation}
\end{lemma}
